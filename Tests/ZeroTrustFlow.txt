# Request without Token -> Blocked by Gateway

Request:
  curl --location 'http://localhost:4000/api/users/'

Response:
  status code:- 401 unauthorised
  Message:- {"error":"Missing token"}



# Bypass Gateway → Blocked by Backend

Request:
  curl --location 'http://localhost:6000/api/users/'

Response:
  status code:- 403 forbidden
  Message:-  "error": "Forbidden: Missing x-verified-user header"
